== 순열
* arr을 (arr.length - 2) 개 뽑는 모든 순열 res 에 추가
* cnt 에 갯수 출력

[source,js]
----
function arrPr(arr, r) {
  let res = [];
  let tmp = [];
  // let cnt = 0;
  let ch = Array.from(Array(arr.length + 1), () => 0);
  function DFS(L) {
    if (L === arr.length) {
      res.push(tmp.join(''));
      // cnt++;
    } else {
      for (let i = 0; i < arr.length; i++) {
        if (ch[i] === 0) {
          tmp[L] = arr[i];
          ch[i] = 1;
          DFS(L + 1);
          ch[i] = 0;
        }
      }
    }
  }
  DFS(arr.length - r);
  return res;
  // return cnt;
}
