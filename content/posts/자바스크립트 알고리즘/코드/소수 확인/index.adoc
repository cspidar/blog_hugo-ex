== 소수 확인

* n을 n의 제곱근 (Math.sqrt(n)) 까지 나눈 나머지가 0이 존재하면 안됨

* n이 소수면 true, 아니면 false 반환

[source,js]
----
const chkPrime = (n) => {
  if (n === 0) return false;
  if (n === 1) return false;
  else if (n === 2) return true;
  else {
    for (let i = 2; i <= Math.sqrt(n); i++) {
      if (n % i === 0) return false;
    }
    return true;
  }
};
----
