== 중복 순열
* arr 에서 중복을 허락하여 r 개 뽑아 res 에 추가
* cnt에 갯수 출력

[source,js]
----
function arrPir(arr, r) {
  let res = [];
  let tmp = [];
  // let cnt = 0;
  function DFS(L) {
    if (L === r) {
      res.push(tmp.slice());
      // cnt++;
    } else {
      for (let i = 0; i < arr.length; i++) {
        // L차 순회하며 각 자리에 v 입력
        tmp[L] = arr[i];
        DFS(L + 1);
      }
    }
  }
  DFS(0);
  // return cnt;
  return res;
}
----
